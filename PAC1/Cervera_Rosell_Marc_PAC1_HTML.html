# Procediment per recablejar el graf
import random as rnd # Generacio nombres aleatoris --> generacio de nombres entre 0.0 i 1.0 per saber si recablejar i per eleccio random de nou node
def rewiring_beta_model(graph, p):
    x = graph.copy() # Bibliografia: [3]
    nodes = x.nodes() # Tots els nodes del graf base # Bibliografia: [4]
    edges = x.edges() # Totes les arestes del graf base # Bibliografia: [5]
    #p = 0.7 # Probabilitat de recablejar
    """
    Passos rewiring:
    1.- Trobar els K/2 veins a la dreta.
    2.- Si hi ha aresta entre el node que s'esta tractant i el ve i la probabilitat (generada random) < p (probabilitat de rewire) -> rewire
        2.1.- El nou node no pot ser ell mateix
        2.2.- El nou node no pot ser un ve que ja te.
    """
    for node in nodes:
        # Step 1 - Buscar els K/2 veins de la dreta
        rightmost_neighbours = [] # Per guardar els veins
        for vei in range(1, K // 2 + 1): # range(1, 4 // 2 + 1) = [1, 2]
            rightmost_neighbours.append((node + vei) % N) # Calcul dels veins de la dreta
            # Step 2 - Si hi ha aresta entre el node que s'esta tractant i els veins i la probabilitat random < probabilitat de rewire
            for neighbour in rightmost_neighbours:
                if x.has_edge(node, neighbour) and round(rnd.uniform(0.0, 1.0), 1) < p: # Bibliografia: [8] [9] [10] [11]
                    new_possible_connections = nodes - x.neighbors(node) - {node} # Els nous possibles nodes son tots - ell mateix - els veins actuals
                    x.remove_edge(node, neighbour) # Descablejar el node i el vei # Bibliografia: [12]
                    new_node = rnd.choice(list(new_possible_connections)) # Bibliografia: [13]
                    x.add_edge(node, new_node) # Bibliografia [14]
        return x

N = 200
K = 4
times_rep = 100 # Nombre de repeticions per cada beta
beta = [round(i, 2) for i in np.arange(0.0, 1.5, 0.05)] # Probabilitats beta de 0.0 a 1.0 amb un salt de 0.05 # Bibliografia: [18]
path_length = [] # Llista per guardar les L a graficar no normalitzades --> Mitjana de cada beta
clustering_coeff = [] # Llista per guardar les C a graficar no normalitzades --> Mitjana de cada beta
for prob in beta:
    paths_every_beta = []
    clustering_coeff_every_beta = []
    for i in range(times_rep):
        graph = nx.watts_strogatz_graph(N, K, prob) # Generacio del graf base
        paths_every_beta.append(nx.average_shortest_path_length(graph)) # Per cada repeticio ens quedem amb la longitud mitjana del cami mes curt
        # Bibliografia: [19]
        clustering_coeff_every_beta.append(nx.average_clustering(graph)) # Per cada repeticio ens quedem amb el coeficient mitja del graf
        # Bibliografia: [20]
    path_length.append(np.mean(paths_every_beta)) # Bibliografia: [21]
    clustering_coeff.append(np.mean(clustering_coeff_every_beta))
path_length_norm= [] # L normalitzades
# Bibliografia: [22]
for i in range(len(path_length)):
    path_length_norm.append((path_length[i] - min(path_length)) / (max(path_length) - min(path_length)))
clustering_coeff_norm = [] # C normalitzades
for j in range(len(clustering_coeff)):
    clustering_coeff_norm.append((clustering_coeff[j] - min(clustering_coeff)) / (max(clustering_coeff) - min(clustering_coeff)))

import random as rnd

# final_graph_nodes = Nombre de nodes amb finals --> Condicio per parar l'algorisme
def barabasi_albert_preferential_attachement(graph, m, final_graph_nodes):
    final_graph = graph.copy() # Bibliografia: [3]
    initial_nodes = len(list(graph.nodes()))
    for i in range(initial_nodes, final_graph_nodes):
        node_degrees = dict(final_graph.degree()) # Diciconari {node: grau} # Bibliografia: [28]
        connections_of_new_node = rnd.choices(list(node_degrees.keys()), weights=list(node_degrees.values()), k=m) # Bibliografia: [29] i [30]
        # weights = possibilitat per cada node. k = llarg de la llista reotrn
        final_graph.add_node(len(list(nx.nodes(final_graph)))) # len(list(nx.graph(nodes))) - Nom node = segent nombre # Bibliografia: [31] i [32]
        for node in connections_of_new_node:
            final_graph.add_edge(node, i)
    return final_graph

Soluci: #Bibliografia: [64]

La imatge s una metfora visual de la consideraci de Kleinberg sobre una xarxa on els nodes s'enllaen de manera aleatrria tenint en compte que la probabilitat d'enlla disminueix amb la distncia. Aquesta probabilitat d'enlla entre nodes segueix una llei de potncia amb un exponent paramtric = 2.

Quan aquest parmetre es exactament 2, el graf/xarxa es troba en equilibri. s a dir, els short-cuts antre dos nodes (a qualsevol distncia) es poden trobar amb poca informaci. Tot i que la probabilitat de connexi disminueix quan ms lluny es va, tamb cal tenir en compte que com ms lluny es va, ms nodes hi ha per connectar-se.

Si > 2, la probabilitat de que dos nodes s'enllain disminueix molt rpidament amb la distncia, de tal manera que noms els nodes propers entre si tenen alguna possibilitat de connectar-se.

Si < 2, el graf presenta el problema que presenta el model beta de Watts-Strogatz. Existeixen camins curts per no es poden trobar amb poca informaci.

En observar la imatge, es pot veure, que aquesta representa la caustica de = 2. s a dir, la 9 avinguda es el nostre punt de vista (el nostre node) i com ms ens allunyem del nostre carrer ms llocs diferents apareixen en el mapa (cada lloc diferent s un node) per com ms lluny mirem des de la 9 avinguda ms es redueix la probabilitat de que ens connectem a aquell lloc on mirem. En altres paraules, s ms fcil (ms probable) que si ens trobem a la 9 avinguda, ens connectem amb la 10 avinguda que amb un altre carrer d'un altre pas.

Per tant, la conclusi extreta de la imatge desprs de la lectura del captol 5, s que la imatge dna resposta a la pregunta que es feia Kleinberg. Com es poden trobar els short-paths del model beta de Watts-Strogatz amb poca informaci?. La pregunta original que es feia Kleinberg no inclou la part de la poca informaci, per com que l'experiment del small-world de Milgram li va servir d'inspiraci, la pregunta es pot completar com s'ha vist.
